git clone (link del repository)
cd (navegar dentro del disco)
dir (ver el contenido de la carpeta en la que estas)
git status (ver el estado de la pagina de git)
mkdir "	" (crear carpeta dentro del directorio)
cd .. (ir a la carpeta anterior)
git add .(agregar todos los archivo a git)
git commit -m "	" (actualizar la pagina de git mas mensaje)
git config --global user.email (para configurar el correo del usuario a agregar el archivo)
git config --global user.name (para configurar el usuario de la persona a agregar el archivo)
git push (finalizar la actualizacion del archivo)
git add site/(nombre del archivo) (para agregar solo el archivo que seleciones)
browser-sync start --server --directory --files "*" (para usar browser-sync)
